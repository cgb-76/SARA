```bash
  test -d raw/slack && echo "raw/slack/ OK" &&
  test -d raw/documents && echo "raw/documents/ OK" &&
  test -d .sara/templates && echo ".sara/templates/ OK" &&
  echo "=== File checks ===" &&
  test -f .sara/config.json && echo ".sara/config.json OK" &&
  test -f pipeline-state.json && echo "pipeline-state.json OK" &&
  test -f wiki/CLAUDE.md && echo "wiki/CLAUDE.md OK" &&
  test -f wiki/index.md && echo "wiki/index.md OK" &&
  test -f wiki/log.md && echo "wiki/log.md OK" &&
  test -f .sara/templates/requirement.md && echo "requirement.md OK" &&

  test -f .sara/templates/action.md && echo "action.md OK" &&
  test -f .sara/templates/risk.md && echo "risk.md OK" &&
  test -f .sara/templates/stakeholder.md && echo "stakeholder.md OK" &&
  echo "=== Content checks ===" &&
  python3 -c "import json; c=json.load(open(\".sara/config.json\")); assert \"verticals\" in c and \"departments\" in c and \"project\" in c and \"schema_version\" in c, \"config.json missing keys\"; print(\".sara/config.json structure OK\")" &&
  python3 -c "import json; s=json.load(open(\"pipeline-state.json\")); assert \"counters\" in s and \"items\" in s; i=s[\"counters\"][\"ingest\"]; e=s[\"counters\"][\"entity\"]; assert all(k in i for k in [\"MTG\",\"EML\",\"SLK\",\"DOC\"]); assert all(k in e for k in [\"REQ\",\"DEC\",\"ACT\",\"RISK\",\"STK\"]); assert all(v==0 for v in list(i.values())+list(e.values())), \"counters not all zero\"; print(\"pipeline-state.json structure OK\")" &&
  grep -q "schema_version" .sara/templates/requirement.md && echo "requirement.md schema_version OK" &&
  grep -q "schema_version" .sara/templates/decision.md && echo "decision.md schema_version OK" &&
  grep -q "schema_version" .sara/templates/action.md && echo "action.md schema_version OK" &&
```

Confirm it outputs an error message containing "Aborting" (or equivalent) and exits without creating
or modifying any files.

```bash
# Before second run, record modification time of config.json
BEFORE=$(stat -c %Y /tmp/sara-test-run/.sara/config.json 2>/dev/null || date +%s)

# Run /sara-init again (should abort)
# ... invoke /sara-init in /tmp/sara-test-run ...

# After second run, check config.json was not modified
AFTER=$(stat -c %Y /tmp/sara-test-run/.sara/config.json 2>/dev/null || date +%s)
[ "$BEFORE" = "$AFTER" ] && echo "Guard clause OK — config.json not modified" || echo "FAIL — config.json was modified by second run"
```

Step 5 — Clean up:
```bash
rm -rf /tmp/sara-test-run
```
